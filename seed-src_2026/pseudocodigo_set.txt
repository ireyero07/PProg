Funcion set_del:
Declaro variables de entrada set e id



Status set_del(Set* set, Id id) {
  int i, pos;

  if (!set || id == NO_ID || set->n_ids >= MAX_IDS) return ERROR;

  pos = set_find(set, id);
  if (pos==-1) {
    return ERROR;
  }

  /* compact array */
  for (i = pos; i < set->n_ids - 1; i++) {
    set->ids[i] = set->ids[i + 1];
  }

  set->n_ids--;

  return OK;
}
